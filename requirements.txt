from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from graph_analyzer import CorruptionGraphAnalyzer

app = FastAPI(title="Corruption Epidemic API", version="1.0.0")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

analyzer = CorruptionGraphAnalyzer()


@app.get("/")
def root():
    return {"message": "Korruptsioner Epidemiyasi API ishlamoqda", "status": "ok"}


@app.post("/employees")
def add_employee(id: str, name: str, risk: float):
    analyzer.add_employee(id, name, risk)
    return {"status": "added", "id": id}


@app.post("/connections")
def add_connection(from_id: str, to_id: str, type: str):
    analyzer.add_connection(from_id, to_id, type)
    return {"status": "connected"}


@app.get("/super-spreaders")
def get_super_spreaders():
    return analyzer.find_super_spreaders(top_n=5)


@app.get("/quarantine/{spreader_id}")
def get_quarantine(spreader_id: str):
    return analyzer.get_quarantine_list(spreader_id)


@app.get("/graph")
def get_graph():
    """Barcha tugunlar va bog'liqliklarni qaytarish (frontend uchun)"""
    nodes = [
        {"id": n, **analyzer.G.nodes[n]}
        for n in analyzer.G.nodes()
    ]
    edges = [
        {"source": u, "target": v, **analyzer.G[u][v]}
        for u, v in analyzer.G.edges()
    ]
    return {"nodes": nodes, "edges": edges}
